CREATE TABLE DASHBOARDGRAFIK.SYSCODE
(
  ID           NUMBER(19)                       
  JENIS_DATA   VARCHAR2(255 BYTE),
  TAHUN        VARCHAR2(255 BYTE),
  BULAN        VARCHAR2(255 BYTE),
  TARGET_RKAP  VARCHAR2(255 BYTE),
  SATUAN       VARCHAR2(255 BYTE),
  TYPE         VARCHAR2(255 BYTE),
  CREATED_AT   TIMESTAMP(6),
  UPDATED_AT   TIMESTAMP(6)
)

CREATE SEQUENCE DASHBOARDGRAFIK.SYSCODE_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.SYSCODE_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.SYSCODE 
FOR EACH ROW
BEGIN
  SELECT SYSCODE_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
