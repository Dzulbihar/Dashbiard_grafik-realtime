CREATE TABLE DASHBOARDGRAFIK.COST_PER_TEUS
(
  JENIS_DATA    CHAR(20 BYTE),
  TAHUN         CHAR(20 BYTE),
  BULAN         CHAR(20 BYTE),
  TOTALBIAYA    CHAR(40 BYTE),
  TOTALTEUS     CHAR(40 BYTE),
  BIAYAPERTEUS  CHAR(40 BYTE)
)

CREATE SEQUENCE DASHBOARDGRAFIK.COST_PER_TEUS_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.COST_PER_TEUS_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.COST_PER_TEUS
FOR EACH ROW
BEGIN
  SELECT COST_PER_TEUS_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
